### 새 노드를 머리부터 삽입
```c
#include <iostream>
using namespace std;

#define MAX_LEN 1000000
typedef char Data;

typedef struct _node
{
	Data data;
	struct _node *next;
	struct _node *prev;
} Node;

int	main(void)
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	char str[MAX_LEN];
	cin >> N;

	while (N--)
	{
		Node *head = (Node*)malloc(sizeof(Node));
		Node *tail = (Node*)malloc(sizeof(Node));

		head->prev = NULL;
		head->next = tail;

		tail->next = NULL;
		tail->prev = head;

		Node *cur = head;

		cin >> str;
		for (int idx = 0; str[idx]; ++idx)
		{
			if (str[idx] == '>')
			{
				if (cur->next != tail)
					cur = cur->next;
			}
			else if (str[idx] == '<')
			{
				if (cur->prev != NULL)
					cur = cur->prev;
			}
			else if (str[idx] == '-')
			{
				if (cur->prev != NULL)
				{
					cur->next->prev = cur->prev;
					cur->prev->next = cur->next;
					cur = cur->prev;
				}
			}
			else
			{
				Node *newNode = (Node*)malloc(sizeof(Node));
				newNode->data = str[idx];

				newNode->next = cur->next;
				cur->next->prev = newNode;

				cur->next = newNode;
				newNode->prev = cur;

				cur = newNode;
			}
		}
		Node *curr = head;
		while (curr->next != tail)
		{
			cout << curr->next->data;
			curr = curr->next;
		}
		cout << '\n';
	}
}
```

### 새 노드를 꼬리부터 삽입

#include <iostream>
using namespace std;

#define MAX_LEN 1000000
typedef char Data;

typedef struct _node
{
	Data data;
	struct _node *next;
	struct _node *prev;
} Node;

int	main(void)
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	char str[MAX_LEN];
	cin >> N;

	while (N--)
	{
		Node *head = (Node*)malloc(sizeof(Node));
		Node *tail = (Node*)malloc(sizeof(Node));

		head->prev = NULL;
		head->next = tail;

		tail->next = NULL;
		tail->prev = head;

		Node *cur = tail;

		cin >> str;
		for (int idx = 0; str[idx]; ++idx)
		{
			if (str[idx] == '>')
			{
				if (cur->prev != head && cur->prev != NULL)
					cur = cur->prev;
			}
			else if (str[idx] == '<')
			{
				if (cur->next != NULL)
					cur = cur->next;
			}
			else if (str[idx] == '-')
			{
				if (cur->next != NULL)
				{
					cur->next->prev = cur->prev;
					cur->prev->next = cur->next;
					cur = cur->next;
				}
			}
			else
			{
				Node *newNode = (Node*)malloc(sizeof(Node));
				newNode->data = str[idx];

				newNode->prev = cur->prev;
				cur->prev->next = newNode;

				newNode->next = cur;
				cur->prev = newNode;
				cur = newNode;
			}
		}
		Node *curr = tail;
		while (curr->prev != head)
		{
			cout << curr->prev->data;
			curr = curr->prev;
		}
		cout << '\n';
	}
}
