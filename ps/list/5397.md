## 풀이
```
번역에 문제가 있다고 한다. "만약 커서의 위치가 줄의 마지막이 아니라면, 그 문자를 입력하고, <커서는 오른쪽으로 한 칸 이동한다.>" 가 아니라
`커서 뒤의 모든 문자가 오른쪽으로 한 칸 이동하는 것`이다.

# 입력
>>>ABC->>e<<<-
>>>ABC->>e<<-
0123456789<a<<b<<<c<<<<d<<<<<e

# 출력
=====
ABe
Be
e012d345c67b8a9
```
### C++ STL list 사용
```c++
#include <iostream>
#include <list>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int N;
    cin >> N;
    while (N--)
    {
        list<char> List;
        string s1;
        list<char>::iterator cursor;

		cursor = List.begin();
        cin >> s1;
        for (int idx = 0; s1[idx]; ++idx)
        {
            if (s1[idx] == '<')
            {
                if (cursor != List.begin())
                    cursor--;
            }
            else if (s1[idx] == '>')
            {
                if (cursor != List.end())
                    cursor++;
            }
            else if (s1[idx] == '-')
            {
                if (cursor != List.begin())
                {
                    cursor--;
                    cursor = List.erase(cursor);
                }
            }
            else 
                List.insert(cursor, s1[idx]);
        }
        for (cursor = List.begin(); cursor != List.end(); ++cursor)
			cout << *cursor;
        cout << '\n';
    }
}
```


### 새 노드를 머리부터 삽입
```c
#include <iostream>
using namespace std;

#define MAX_LEN 1000000
typedef char Data;

typedef struct _node
{
	Data data;
	struct _node *next;
	struct _node *prev;
} Node;

int	main(void)
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	char str[MAX_LEN];
	cin >> N;

	while (N--)
	{
		Node *head = (Node*)malloc(sizeof(Node));
		Node *tail = (Node*)malloc(sizeof(Node));

		head->prev = NULL;
		head->next = tail;

		tail->next = NULL;
		tail->prev = head;

		Node *cur = head;

		cin >> str;
		for (int idx = 0; str[idx]; ++idx)
		{
			if (str[idx] == '>')
			{
				if (cur->next != tail)
					cur = cur->next;
			}
			else if (str[idx] == '<')
			{
				if (cur->prev != NULL)
					cur = cur->prev;
			}
			else if (str[idx] == '-')
			{
				if (cur->prev != NULL)
				{
					cur->next->prev = cur->prev;
					cur->prev->next = cur->next;
					cur = cur->prev;
				}
			}
			else
			{
				Node *newNode = (Node*)malloc(sizeof(Node));
				newNode->data = str[idx];

				newNode->next = cur->next;
				cur->next->prev = newNode;

				cur->next = newNode;
				newNode->prev = cur;

				cur = newNode;
			}
		}
		Node *curr = head;
		while (curr->next != tail)
		{
			cout << curr->next->data;
			curr = curr->next;
		}
		cout << '\n';
	}
}
```

### 새 노드를 꼬리부터 삽입
```c++
#include <iostream>
using namespace std;

#define MAX_LEN 1000000
typedef char Data;

typedef struct _node
{
	Data data;
	struct _node *next;
	struct _node *prev;
} Node;

int	main(void)
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int N;
	char str[MAX_LEN];
	cin >> N;

	while (N--)
	{
		Node *head = (Node*)malloc(sizeof(Node));
		Node *tail = (Node*)malloc(sizeof(Node));

		head->prev = NULL;
		head->next = tail;

		tail->next = NULL;
		tail->prev = head;

		Node *cur = tail;

		cin >> str;
		for (int idx = 0; str[idx]; ++idx)
		{
			if (str[idx] == '>')
			{
				if (cur->prev != head && cur->prev != NULL)
					cur = cur->prev;
			}
			else if (str[idx] == '<')
			{
				if (cur->next != NULL)
					cur = cur->next;
			}
			else if (str[idx] == '-')
			{
				if (cur->next != NULL)
				{
					cur->next->prev = cur->prev;
					cur->prev->next = cur->next;
					cur = cur->next;
				}
			}
			else
			{
				Node *newNode = (Node*)malloc(sizeof(Node));
				newNode->data = str[idx];

				newNode->prev = cur->prev;
				cur->prev->next = newNode;

				newNode->next = cur;
				cur->prev = newNode;
				cur = newNode;
			}
		}
		Node *curr = tail;
		while (curr->prev != head)
		{
			cout << curr->prev->data;
			curr = curr->prev;
		}
		cout << '\n';
	}
}
```
